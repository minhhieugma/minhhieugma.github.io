    [Test]
    public void SecondRendersCorrectly()
    {
    // Arrange
        using var ctx = new TestContext();
        var mock = ctx.Services.AddMockHttpClient();

        mock.When("https://api.openweathermap.org/data/2.5/weather?lat=43.7615&lon=-79.4111&appid=bbd1ee00f3a4026cc99e112e4fbb7a82&units=metric")
            .RespondJson(new
            {
                main = new
                {
                    temp = 21.91
                },
                name = "North York"
            });

    // Act
        var cut = ctx.RenderComponent<Weather>();
        cut.WaitForElement("div.country");

    // Assert
        Assert.AreEqual(6, cut.FindAll("div").Count);
        Assert.AreEqual("North York", cut.Find("div.country").TextContent);
    }
}
